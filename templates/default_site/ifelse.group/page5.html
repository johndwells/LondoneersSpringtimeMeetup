ifElse: Early parsing of advanced conditionals in EE templates.

{exp:ifelse parse="inward"}
	{if member_id == '1' OR group_id == '2'}
		Admin content
	{if:elseif logged_in}
		Member content
	{if:else}
		Public content
	{/if}
{/exp:ifelse}
